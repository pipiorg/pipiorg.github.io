222 internal List<PiPiCSharpField> ExtractField()
223 {
224 try
225 {
226 var fields = new List<PiPiCSharpField>();
227
228 IntPtr cFields = PiPiExtractorExtractField(this.cExtractor);
229
230 int cFieldSize = PiPiExtractorExtractedFieldsSize(cFields);
231 for (int i = 0; i < cFieldSize; i++)
232 {
233 IntPtr cField = PiPiExtractorGetExtractedField(cFields, i);
234
235 string name = PiPiExtractorGetExtractedFieldName(cField);
236 string fontName = PiPiExtractorGetExtractedFieldFontName(cField);
237
238 float fontSize = PiPiExtractorGetExtractedFieldFontSize(cField);
239
240 ushort pageIndex = PiPiExtractorGetExtractedFieldPageIndex(cField);
241 ushort cType = PiPiExtractorGetExtractedFieldType(cField);
242
243 double width = PiPiExtractorGetExtractedFieldWidth(cField);
244 double height = PiPiExtractorGetExtractedFieldHeight(cField);
245 double x = PiPiExtractorGetExtractedFieldX(cField);
246 double y = PiPiExtractorGetExtractedFieldY(cField);
247
248 PiPiCSharpFieldType type = PiPiCSharpConstants.FieldTypeInvertMap[cType];
249
250 PiPiCSharpField field = new PiPiCSharpField(name, type, pageIndex, x, y, width, height, fontName, fontSize);
251 fields.Add(field);
252
253 DeletePiPiExtractorExtractedField(cField);
254 }
255
256 DeletePiPiExtractorExtractedFields(cFields);
257
258 return fields;
259 }
260 catch (Exception e)
261 {
262 throw new PiPiCSharpExtractException(PiPiCSharpExtractException.PiPiCSharpExtractExceptionCode.Unknown, e);
263 }
264 }
